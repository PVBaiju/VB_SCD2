from pyspark.sql.functions import col, current_date, lit, row_number
from pyspark.sql.window import Window
from delta.tables import DeltaTable

# =======================================
# Step 1: Get latest file path dynamically
# =======================================
folders = dbutils.fs.ls("/mnt/source/fact_scd1/")
latest_year = max([f.name.replace('/', '') for f in folders])

folders = dbutils.fs.ls(f"/mnt/source/fact_scd1/{latest_year}/")
latest_month = max([f.name.replace('/', '') for f in folders])

folders = dbutils.fs.ls(f"/mnt/source/fact_scd1/{latest_year}/{latest_month}/")
latest_day = max([f.name.replace('/', '') for f in folders])

latest_path = f"/mnt/source/fact_scd1/{latest_year}/{latest_month}/{latest_day}/*.parquet"
print(f"Reading from: {latest_path}")

# Construct file_date from folder

file_date = f"{latest_year}-{latest_month}-{latest_day}"

# =======================================
# Step 2: Read raw data and enrich
# =======================================
raw_df = spark.read.parquet(latest_path)

spark.conf.set("spark.databricks.delta.schema.autoMerge.enabled", "true")

enriched_df = (raw_df
               .withColumn("ingest_date", current_date())   # actual load date
               .withColumn("file_date", lit(file_date)))    # folder date

# =======================================
# Step 3: Deduplicate (based on business key)
# =======================================
# Use ORDER_ID plus(+) ORDER_ITEM_ID for order_items fact.
window = Window.partitionBy("ORDER_ID", "ORDER_ITEM_ID") \
               .orderBy(col("LOAD_TS").desc())

dedup_df = (enriched_df
            .withColumn("rn", row_number().over(window))
            .filter(col("rn") == 1)
            .drop("rn"))

# =======================================
# Step 4: Merge into Delta (SCD1 logic)
# =======================================
delta_path = "/mnt/curated/order_delta_scd1"
table_name = "order_delta_scd1"

if not DeltaTable.isDeltaTable(spark, delta_path):
    # Initial load
    # Partition for performance
    (dedup_df.write
        .format("delta")
        .mode("overwrite")
        .option("mergeSchema", "true")
        .partitionBy("file_date")     
        .save(delta_path))
    
    spark.sql(f"""
        CREATE TABLE IF NOT EXISTS {table_name}
        USING DELTA
        LOCATION '{delta_path}'
    """)
else:
    # SCD1 Merge (update if match, insert if not)
    deltaTable = DeltaTable.forPath(spark, delta_path)
    (
        deltaTable.alias("t")
        .merge(
            dedup_df.alias("s"),
            "t.ORDER_ID = s.ORDER_ID AND t.ORDER_ITEM_ID = s.ORDER_ITEM_ID"
        )
        .whenMatchedUpdateAll()
        .whenNotMatchedInsertAll()
        .execute()
    )

# =======================================
# Step 5: Reload and Validate
# =======================================
updated_df = spark.read.format("delta").load(delta_path)

print("Before Merge:")
raw_df.show(60, truncate=False)

print("After Merge (Dedup + SCD1 Applied):")
updated_df.show(60, truncate=False)

from pyspark.sql.functions import current_date, current_timestamp, lit, expr, col, row_number
from pyspark.sql.window import Window
from delta.tables import DeltaTable

# ===================================================
# Step 1: Get latest file path (from ADLS source)
# ===================================================
folders = dbutils.fs.ls("/mnt/source/dim_scd2")
latest_year = max([f.name.replace('/', '') for f in folders])

folders = dbutils.fs.ls(f"/mnt/source/dim_scd2/{latest_year}/")
latest_month = max([f.name.replace('/', '') for f in folders])

folders = dbutils.fs.ls(f"/mnt/source/dim_scd2/{latest_year}/{latest_month}/")
latest_day = max([f.name.replace('/', '') for f in folders])

latest_path = f"/mnt/source/dim_scd2/{latest_year}/{latest_month}/{latest_day}/*.parquet"
print(f"Reading from: {latest_path}")

file_date = f"{latest_year}-{latest_month}-{latest_day}"

# ===================================================
# Step 2: Read source data
# ===================================================
raw_df = spark.read.parquet(latest_path).withColumn("updated_at", current_timestamp())

# Deduplicate incoming batch (order_id & payment_sequential)
window = Window.partitionBy("order_id", "payment_sequential").orderBy(col("updated_at").desc())
dedup_df = raw_df.withColumn("rn", row_number().over(window)) \
                 .filter(col("rn") == 1) \
                 .drop("rn")

# Add SCD2 metadata
incoming_df = (
    dedup_df
    .withColumn("payment_dim_id", expr("uuid()"))
    .withColumn("is_active", lit(1))
    .withColumn("start_date", current_date())
    .withColumn("end_date", lit(None).cast("date"))
    .withColumn("file_date", lit(file_date))
    .withColumn("ingest_date", current_date())
    .withColumn("updated_at", current_timestamp())
)

# ===================================================
# Step 3: Delta table setup
# ===================================================
delta_path = "/mnt/curated/payment_delta_scd2"
table_name = "dim_order_payments_scd2"

if not DeltaTable.isDeltaTable(spark, delta_path):
    # Initial load with SCD2 schema
    (incoming_df.write
        .format("delta")
        .mode("overwrite")
        .option("mergeSchema", "true")
        .save(delta_path))
    
    # Also register metastore so ALTER works later
    spark.sql(f"""
        CREATE TABLE IF NOT EXISTS {table_name}
        USING DELTA
        LOCATION '{delta_path}'
    """)
else:
    deltaTable = DeltaTable.forPath(spark, delta_path)
    df_existing = deltaTable.toDF()

    # ===================================================
    # Step 4: Ensure schema has SCD2 columns ***check***
    # ===================================================
    expected_cols = {
        "payment_dim_id": lit(None).cast("string"),
        "is_active": lit(1).cast("int"),
        "start_date": current_date(),
        "end_date": lit(None).cast("date"),
        "file_date": lit("1900-01-01"),
        "ingest_date": current_date(),
        "updated_at": current_timestamp()
    }

    for col_name, default_expr in expected_cols.items():
        if col_name not in df_existing.columns:
            print(f"!!! Adding column --> {col_name}")
            df_existing = df_existing.withColumn(col_name, default_expr)

    # overwrite table with schema-upgraded version
    (df_existing.write
        .format("delta")
        .mode("overwrite")
        .option("overwriteSchema", "true")
        .save(delta_path))

    deltaTable = DeltaTable.forPath(spark, delta_path)

    # ===================================================
    # Step 5: Identify new or changed rows / (col("t.payment_type").isNull()) ---> for check new record
    # ===================================================
    active_df = deltaTable.toDF().filter(col("is_active") == 1)

    changes_df = (
        incoming_df.alias("s")
        .join(
            active_df.alias("t"),
            (col("s.order_id") == col("t.order_id")) &
            (col("s.payment_sequential") == col("t.payment_sequential")),
            "left_outer"
        )
        .filter(
            (col("t.payment_type").isNull()) | 
            (col("t.payment_type") != col("s.payment_type")) |
            (col("t.payment_installments") != col("s.payment_installments")) |
            (col("t.payment_value") != col("s.payment_value"))
        )
        .select("s.*")
    )

    # Deduplicate changes
    window_changes = Window.partitionBy("order_id", "payment_sequential").orderBy(col("updated_at").desc())
    changes_df_dedup = (
        changes_df.withColumn("rn", row_number().over(window_changes))
                  .filter(col("rn") == 1)
                  .drop("rn")
    )

    # ===================================================
    # Step 6a: Expire old records
    # ===================================================
    (
        deltaTable.alias("t")
        .merge(
            changes_df_dedup.alias("s"),
            "t.order_id = s.order_id AND t.payment_sequential = s.payment_sequential AND t.is_active = 1"
        )
        .whenMatchedUpdate(
            set={
                "is_active": lit(0),
                "end_date": current_date(),
                "updated_at": current_timestamp()
            }
        )
        .execute()
    )

    # ===================================================
    # Step 6b: Insert new/changed rows
    # ===================================================
    (
        deltaTable.alias("t")
        .merge(
            changes_df_dedup.alias("s"),
            "t.order_id = s.order_id AND t.payment_sequential = s.payment_sequential AND t.is_active = 0"
        )
        .whenNotMatchedInsert(
            values={c: col(f"s.{c}") for c in incoming_df.columns}
        )
        .execute()
    )

# ===================================================
# Step 7: Result Check
# ===================================================
print("Ingress Data:")
raw_df.show(50, truncate=False)

print("Dimension table after SCD2 merge:")
updated_df = spark.read.format("delta").load(delta_path)
updated_df.orderBy("order_id", "payment_sequential", "start_date").show(50, truncate=False)
